function renderUsers(){ const users = JSON.parse(localStorage.getItem('users') || '{}'); const history = JSON.parse(localStorage.getItem('quizHistory') || '{}'); const keyword = document.getElementById("search").value.trim(); const listDiv = document.getElementById('userList'); listDiv.innerHTML = ''; const userArray = Object.keys(users) .filter(u => u !== "admin") .map(username => { let total = 0; let correct = 0; if(history[username]){ history[username].forEach(h=>{ total += h.total; correct += h.score; }); } let rate = total ? Math.round((correct/total)*100) : 0; return { username, user: users[username], rate }; }) .sort((a,b)=> b.rate - a.rate); let rank = 1; userArray.forEach(obj => { const username = obj.username; const user = obj.user; const rate = obj.rate; // 🔍 検索 if(keyword !== ""){ if(!user.username.includes(keyword) && !user.fullname.includes(keyword)){ return; } } // 🏆 メダル let medal = ""; if(rank === 1) medal = "🥇"; else if(rank === 2) medal = "🥈"; else if(rank === 3) medal = "🥉"; // 🎨 色分け let color = "#4caf50"; if(rate < 60) color = "#f44336"; else if(rate < 80) color = "#ff9800"; const div = document.createElement('div'); div.className = 'card'; div.innerHTML = `

${rank}位 ${medal}

${user.memberId} | ${user.username} | ${user.fullname}

正答率：${rate}%

`; listDiv.appendChild(div); rank++; }); }
