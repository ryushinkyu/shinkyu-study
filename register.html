document.getElementById('registerBtn').addEventListener('click', ()=>{ const fullname = document.getElementById('fullname').value.trim(); const kana = document.getElementById('kana').value.trim(); const birthdate = document.getElementById('birthdate').value; const email = document.getElementById('email').value.trim(); const password = document.getElementById('password').value; const username = document.getElementById('username').value.trim(); // ===== 入力チェック ===== if(!fullname || !kana || !birthdate || !email || !password || !username){ alert("全て入力してください"); return; } if(password.length < 4){ alert("パスワードは4文字以上にしてください"); return; } // ===== ユーザー存在チェック ===== db.collection("users").doc(username).get().then(doc => { if(doc.exists){ alert("このユーザー名は既に使われています"); return; } // ===== Firebase保存 ===== db.collection("users").doc(username).set({ fullname: fullname, kana: kana, birthdate: birthdate, email: email, password: password, memberId: String(Math.floor(100 + Math.random() * 900)), // 3桁ID username: username }).then(()=>{ // ログイン状態にする localStorage.setItem("currentUser", username); // ホームへ location.href = "home.html"; }).catch(err=>{ console.error(err); alert("登録エラー"); }); }); });
